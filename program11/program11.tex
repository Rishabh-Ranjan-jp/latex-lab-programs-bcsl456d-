\documentclass{article}
\usepackage[linesnumbered,ruled,vlined]{algorithm2e}
\begin{document}
	\begin{algorithm}
		\SetAlgoLined
		\caption{Algorithm for adding two numbers}
		\KwData{Two numbers, \(a\) and \(b\)}
		\KwResult{Sum of \(a\) and \(b\), stored in c}
		\textbf{Input:} \(a, b\) \;
		\textbf{Output:} \(c\) \;
		Read \(a\), \(b\) \;
		\(c \leftarrow a + b\) \;
		Print \(c\) \;
	\end{algorithm}
